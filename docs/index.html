def create_opensime_model():
    model = Model()
        
    model.set_var_default(0)
    model.set_param_default(0)
    #Creates and documents series
    model.var('b_ukuk_d', desc='Bills issued by the UK acquired by the UK: demand')
    model.var('b_ukuk_s', desc='Bills issued by the UK acquired by the UK: supply')
    model.param('b_cb_ukuk_d', desc='Bills issued by the UK, demanded by the UK Central bank') #Error in the equation:b_cb_ukuk_d : under-specified : variable does not have equation
    model.var('b_cb_ukuk_s', desc='Bills issued by the UK, supplied to the UK Central bank')
    model.var('b_uk_s', desc='Bills issued by the UK - total supply')
    model.var('b_ukus_d', desc='Bills issued by the US acquired by the UK: demand')
    model.var('b_ukus_s', desc='Bills issued by the US acquired by the UK: supply')
    model.var('b_usuk_d', desc='Bills issued by the UK acquired by the US: demand')
    model.var('b_usuk_s', desc='Bills issued by the UK acquired by the US: supply')
    model.var('b_us_s', desc='Bills issued by the US - total supply')
    model.var('b_usus_d', desc='Bills issued by the US acquired by the US: demand')
    model.var('b_usus_s', desc='Bills issued by the US acquired by the US: supply')
    model.param('b_cb_usus_d', desc='Bills issued by the US demanded by the US Central Bank') # b_cb_usus_d : under-specified : variable does not have equation
    model.var('b_cb_usus_s', desc='Bills issued by the US supplied to the US Central Bank')
    model.var('cab_uk', desc='Current account balance in the UK')
    model.var('cab_us', desc='Current account balance in the US')
    model.var('cons_uk', desc='Consumption in the UK')
    model.var('cons_us', desc='Consumption in the US')
    model.var('f_cb_uk', desc='Profits of Central Bank in the UK')
    model.var('f_cb_us', desc='Profits of Central Bank in the US')
    model.param('g_uk', desc='Government expenditure in the UK') # No eviews as var exogenas podem ser tratadas como var. No Phyton so parmetros (no tem equaes para elas)
    model.param('g_us', desc='Government expenditure in the US') # No eviews as var exogenas podem ser tratadas como var. No Phyton so parmetros (no tem equaes para elas)
    model.var('h_uk_s', desc='Supply of the UK cash')
    model.var('h_us_s', desc='Supply of the US cash')
    model.var('im_uk', desc='Imports of the UK from the US')
    model.var('im_us', desc='Imports of the US from the UK')
    model.param('kab_uk', desc='Capital account balance in the UK') # variavel sem equacao (indefinida)
    model.param('kab_us', desc='Current account balance in the US') # variavel sem equacao (indefinida)
    model.var('psbr_uk', desc='Government deficit in the UK')
    model.var('psbr_us', desc='Government deficit in the US')
    model.param('r_uk', desc='Interest rate on the UK bills') #variveis exogenas so tratadas como parmetros (no h equaes para elas)
    model.param('r_us', desc='Interest rate on the US bills') #variveis exogenas so tratadas como parmetros (no h equaes para elas)
    model.var('t_uk', desc='Tax revenue in the UK')
    model.var('t_us', desc='Tax revenue in the US')
    model.var('v_uk', desc='Net financial assets of the UK')
    model.var('v_us', desc='Net financial assets of the US')
    model.var('x_uk', desc='Exports from the UK to the US')
    model.var('x_us', desc='Exports from the US to the UK') ###########
    model.var('xr_uk', desc='Exchange rate: units of dollars against 1 unit of pound') #################
    model.var('xr_us', desc='Exchange rate: units of pounds against 1 unit of dollar')
    model.var('yd_uk', desc='Disposable income in the UK')
    model.var('yd_us', desc='Disposable income in the US')
    model.param('ydhs_uk', desc='Haig-Simons disposable income in the UK') # under-specified : variable does not have equation
    model.param('ydhs_us', desc='Haig-Simons disposable income in the US') #under-specified : variable does not have equation
    model.var('y_uk', desc='Income in the UK')
    model.var('y_us', desc='Income in the US')
    model.var('h_uk_h', desc='Holding of money in Uk')
    model.var('h_us_h', desc='Holding of money in Us')
    model.var('tb_uk', desc='Trade balance UK')
    model.var('tb_us', desc='Trade balance US')
    model.var('bp_uk', desc='UK Balance of payment')
    model.var('bp_us', desc='US Balance of payment')
    model.var('yd_hs_uk', desc='...')
    model.var('yd_hs_us', desc='...')
    model.var('nafa_uk', desc='...')
    model.var('nafa_us', desc='...')
    model.var('kabp_uk', desc='...')
    model.var('kabp_us', desc='...')



    #Generate parameters
    model.param('alpha1_us', desc='Propensity to consume out of income the US')
    model.param('alpha2_us', desc='Propensity to consume out of wealth the US')
    model.param('alpha1_uk', desc='Propensity to consume out of income the UK')
    model.param('alpha2_uk', desc='Propensity to consume out of wealth the UK')
    model.param('eps0', desc='Parameter determining real exports in the UK')
    model.param('eps1', desc='Parameter determining real exports in the UK')
    model.param('eps2', desc='Parameter determining real exports in the UK')
    model.param('lambda10', desc='Parameter in asset demand function')
    model.param('lambda11', desc='Parameter in asset demand function')
    model.param('lambda12', desc='Parameter in asset demand function')
    model.param('lambda20', desc='Parameter in asset demand function')
    model.param('lambda21', desc='Parameter in asset demand function')
    model.param('lambda22', desc='Parameter in asset demand function')
    model.param('lambda40', desc='Parameter in asset demand function')
    model.param('lambda41', desc='Parameter in asset demand function')
    model.param('lambda42', desc='Parameter in asset demand function')
    model.param('lambda50', desc='Parameter in asset demand function')
    model.param('lambda51', desc='Parameter in asset demand function')
    model.param('lambda52', desc='Parameter in asset demand function')
    model.param('mu0', desc='Parameter determining real imports in the UK')
    model.param('mu1', desc='Parameter determining real imports in the UK')
    model.param('mu2', desc='Parameter determining real imports in the UK')
    model.param('theta_uk', desc='Tax rate in the UK')
    model.param('theta_us', desc='Tax rate in the US')


    #equations

    #    model opensime

    #1. INCOME AND WEALTH
    
    #Disposable income in UK - eq. 1
    model.add('yd_uk = (y_uk + r_uk(-1)*b_ukuk_s(-1) + xr_us*r_us(-1)*b_ukus_s(-1))*(1 - theta_uk)')
          
    #Haig-Simons disposable income in UK - eq. 2
    model.add('yd_hs_uk = yd_uk + d(xr_us)*b_ukus_s(-1)')
          
    #Wealth accumulation in UK - eq. 3
    model.add('v_uk = v_uk(-1) + yd_hs_uk - cons_uk')
          
    #Disposable income in US - eq. 4
    model.add('yd_us = (y_us + r_us(-1)*b_usus_s(-1) + xr_uk*r_uk(-1)*b_usuk_s(-1))*(1 - theta_us)')
          
    #Haig-Simons disposable income in US - eq. 5
    model.add('yd_hs_us = yd_us + d(xr_uk)*b_usuk_s(-1)')

    #Wealth accumulation in US - eq. 6
    model.add('v_us = v_us(-1) + yd_hs_us - cons_us')

    #Taxes in UK - eq. 7
    model.add('t_uk = theta_uk*(y_uk + r_uk(-1)*b_ukuk_d(-1) + xr_us*r_us(-1)*b_ukus_s(-1))')

    #Taxes in US - eq. 8
    model.add('t_us = theta_us*(y_us + r_us(-1)*b_usus_d(-1) + xr_uk*r_uk(-1)*b_usuk_s(-1))')

    #Income UK - eq. 9
    model.add('y_uk = cons_uk + g_uk + x_uk - im_uk')

    #Income US - eq. 10
    model.add('y_us = cons_us + g_us + x_us - im_us')

    #consumption in UK - eq. 11
    model.add('cons_uk = alpha1_uk* yd_hs_uk + alpha2_uk*v_uk(-1)')

    #consumption in US - eq. 12
    model.add('cons_us = alpha1_us* yd_hs_us + alpha2_us*v_us(-1)')

    #2. TRADE

    #Exports from UK - eq. 13
    model.add('x_uk = exp(eps0 - eps1*log(xr_uk(-1)) + eps2*log(y_us))')

    #Imports of UK - eq. 14
    model.add('im_uk = exp(mu0 + mu1*log(xr_uk(-1)) + mu2*log(y_uk))')

    #Exports of US - eq. 15
    model.add('x_us = im_uk*xr_uk')

    #Imports of US - eq. 16
    model.add('im_us = x_uk*xr_uk')

    #3. ASSET DEMANDS

    #Demand for UK bills in UK - eq. 17
    model.add('b_ukuk_d = v_uk*(lambda10 + lambda11*r_uk - lambda12* r_us)')

    #Demand for US bills in UK - eq. 18
    model.add('b_ukus_d = v_uk*(lambda20 - lambda21*r_uk + lambda22*r_us)')

    #Demand for US	bills in US - eq. 19
    model.add('b_usus_d = v_us*(lambda40 + lambda41*r_us - lambda42*r_uk)')

    #Demand for UK bills in US - eq. 20
    model.add('b_usuk_d = v_us*(lambda50 - lambda51*r_us + lambda52*r_uk)')

    #Holding of money in UK - eq. 21
    model.add('h_uk_h = v_uk - b_ukuk_s - (b_ukus_s*xr_us)')

    #Holding of money in US - eq. 22
    model.add('h_us_h = v_us - b_usus_s - (b_usuk_s*xr_uk)')

    #4. ASSET SUPPLIES

    #Supply of UK bills to UK households - eq. 23
    model.add('b_ukuk_s = b_ukuk_d')

    #Supply of US bills to US households - eq. 24
    model.add('b_usus_s = b_usus_d')

    #Suplly of UK bills to US households - eq. 25
    model.add('b_usuk_s = b_usuk_d*xr_us')

    #Supply of US bills to UK households - eq. 26
    model.add('b_ukus_s = b_ukus_d*xr_uk')

    #5. PUBLIC SECTOR

    #Supply of US bills to US Central bank - eq. 27 (central bank lander of last resort)
    model.add('b_cb_usus_s = b_us_s - b_usus_s - b_ukus_s')

    #Supply of UK bills to UK Central bank - eq. 28
    model.add('b_cb_ukuk_s = b_uk_s - b_ukuk_s - b_usuk_s')

    #Suply of cash in UK - eq. 29
    model.add('h_uk_s = b_cb_ukuk_s')

    #Suply of cash in US - eq. 30
    model.add('h_us_s = b_cb_usus_s')

    #Profits of Central Bank in UK - eq. 31
    model.add('f_cb_uk = r_uk(-1)*b_cb_ukuk_s(-1)') 

    #Profits of Central Bank in US - eq. 32
    model.add('f_cb_us = r_us(-1)*b_cb_usus_s(-1)')

    #Government budget constraint - UK - eq. 33
    model.add('b_uk_s = b_uk_s(-1) + g_uk + r_uk(-1)*b_uk_s(-1) - t_uk - f_cb_uk')

    #Government budget constraint - US - eq. 34
    model.add('b_us_s = b_us_s(-1) + g_us + r_us(-1)*b_us_s(-1) - t_us - f_cb_us')

    #6. EXCHANGE RATES 

    #UK Exchange rate - eq. 35
    #model.add('xr_us = 1/xr_uk') ################
    model.add('xr_us =  (r_uk(-1)*b_usuk_s(-1) - d(b_usuk_s) - x_uk + im_uk)/(r_us(-1)*b_ukus_s(-1) - d(b_ukus_s))')

    #US exchange rate - eq. 36
    #model.add('xr_uk = b_ukus_s/b_ukus_d')########
    model.add('xr_uk = 1/xr_us')

    #7. REDUNDANT EQUATIONS 

    #model.add('h_uk_s = h_uk_h') #- eq. 37
    #model.add('h_us_s = h_us_h') #- eq. 38

    #Equations non essential for the     model (useful measures/indicators) 

    #Government deficit in the UK - eq. 39
    model.add('psbr_uk = g_uk + r_uk(-1)*b_uk_s(-1) - t_uk - f_cb_uk')

    #Government deficit in the US - eq. 40
    model.add('psbr_us = g_us + r_us(-1)*b_us_s(-1) - t_us - f_cb_us')

    #Net accumulation of financial assets in the UK - eq. 41
    model.add('nafa_uk = psbr_uk + cab_uk')

    #Net accumulation of financial assets in the US - eq. 42
    model.add('nafa_us = psbr_us + cab_us')

    #Current account balance - UK - eq. 43
    model.add('cab_uk = x_uk - im_uk + xr_us*r_us(-1)*b_ukus_s(-1) - r_uk(-1)*b_usuk_s(-1)')

    #Current account balance in US - eq. 44
    model.add('cab_us = x_us - im_us + xr_uk*r_uk(-1)*b_usuk_s(-1) - r_us(-1)*b_ukus_s(-1)')

    #Financial account balance in UK - eq. 45
    model.add('kabp_uk = - d(b_ukus_s)*xr_us + d(b_usuk_s)')

    #Financial account balance in US - eq. 46
    model.add('kabp_us = - d(b_usuk_s)*xr_uk + d(b_ukus_s)')

    #Trade balance UK - eq. 47
    model.add('tb_uk =  x_uk - im_uk')

    #Trade balance US - eq. 48
    model.add('tb_us = x_us - im_us')

    #Balance of payment UK - eq. 49
    model.add('bp_uk = cab_uk + kabp_uk')

    #Balance of payment US - eq. 50
    model.add('bp_us = cab_us + kabp_us')

    return    model

# encontrar (t) onde o sistema se equilibra
steady_state = create_opensime_model()
steady_state.set_values(opensime_parameters)
steady_state.set_values(opensime_exogenous)


for i in range(300):
    steady_state.solve(iterations=200, threshold=1e-4, method='broyden')
    prev_soln = steady_state.solutions[-2]
    soln = steady_state.solutions[-1]
    if is_close(prev_soln, soln, atol=1e-4):
        break
